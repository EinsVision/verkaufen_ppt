void drawRect(Mat& img, vector< Rect >& v_rect) {
 
    for (auto it : v_rect) {
        rectangle(img, it, CV_RGB(255, 0, 0), 2);
    }
}
 
int main()
{
    int TH1 = 40;
 
    float min_window_area = 500;
 
    Mat element = getStructuringElement(MORPH_RECT, Size(3, 3), Point(1, 1)); // mor. mask
 
    Mat frame;
    Mat old_frame;
    Mat sub_frame;
    Mat binary_frame, binary_old_frame;
    VideoCapture Pedestrian;
    Pedestrian.open("Pedestrian.mp4");
 
    if (!Pedestrian.isOpened()) // If the previous call to VideoCapture constructor or VideoCapture::open succeeded, the method returns true.
    { 
        cout << "cannot read video";
        return -1;
    }
 
    while (1)
    {
 
        if (!(Pedestrian.read(frame))) //get one frame form video   
            break;
 
        if (old_frame.empty())
        {
            old_frame = frame.clone();
            continue;
        }
 
        cvtColor(frame, binary_frame, COLOR_BGR2GRAY);
        cvtColor(old_frame, binary_old_frame, COLOR_BGR2GRAY);
        //subtract(old_frame, frame, sub_frame);
        absdiff(binary_old_frame, binary_frame, sub_frame);
        threshold(sub_frame, sub_frame, TH1, 255, THRESH_BINARY);
 
        morphologyEx(sub_frame, sub_frame, MORPH_CLOSE, element);
 
        //find contour
        vector< vector< Point> > contours;
        vector< Vec4i> hierarchy;
        findContours(sub_frame.clone(), contours, hierarchy, RETR_CCOMP, CHAIN_APPROX_SIMPLE);
        //drawContours(frame, contours, -1, CV_RGB(255, 0, 0), 5, 8, hierarchy);
 
        //Blob labeling
        vector< Rect > v_rect;
        for (auto it : contours) {
 
            double area = contourArea(it, false);
            if (area > min_window_area)
            {
                Rect mr = boundingRect(Mat(it));
                v_rect.push_back(mr);
                //printf("%lf\n", area);
            }
        }
        drawRect(frame, v_rect);
 
        
        if (v_rect.size() > 0)
        {
            char str[255];
            struct tm now;
            time_t rawtime = time(0);
            localtime_s(&now, &rawtime);
 
            sprintf_s(str, "Y%d_m%d_d%d_h%d_m%d_s%d.jpg", 
                now.tm_year, now.tm_mon, now.tm_mday, now.tm_hour, now.tm_min, now.tm_sec);
            imwrite(str, frame);
        }
        
        imshow("frame", frame);
        imshow("sub_frame", sub_frame);
 
        //old_frame = frame.clone();
 
        if (waitKey(1) >= 0) // control frame speed
            break;
    }
 
    return 0;
}
